    score_a = 0
    score_b = 0
    for i in range(len(a)):
        if a[i] > b[i]:
            score_a += 1
        elif a[i] < b [i]:

3. A Very Big Sum

{% highlight js %}
def aVeryBigSum(ar):
    sum = 0
    for i in range(len(ar)):
        sum += ar[i]

    return (sum)
{% endhighlight %}
