- Staircase
{% highlight js %}
def staircase(n):
    for i in range(1, n+1):
        print(' ' * (n-i) + '#' * (i))
{% endhighlight %}

- Mini-Max Sum
{% highlight js %}
def miniMaxSum(arr):
    sum = 0
    arr.sort()
    for i in range(len(arr)):
        sum += arr[i]
    
    min_sum = sum - arr[4]
    max_sum = sum - arr[0]
    print(min_sum, max_sum)
{% endhighlight %}

- Birthday Cake Candles
{% highlight js %}
def birthdayCakeCandles(ar):
    count = 0
    
    for i in range(len(ar)):
        if ar[i] == max(ar):
            count += 1
    
    return count
{% endhighlight %}
